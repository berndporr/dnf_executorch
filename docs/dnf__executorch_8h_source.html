149 float filter(const float signal, const float noise)
150 {
151 const float delayed_signal = signal_delayLine.process(signal);
152 delayedSignalTensor->mutable_data_ptr<float>()[0] = delayed_signal;
153
154 noise_delayLine.process(noise);
155
156 for (int i = 0; i < noiseDelayLineLength; i++)
157 {
158 noiseTimeSeries->mutable_data_ptr<float>()[i] = noise_delayLine.get(i);
159 }
160
161 const auto &results = trainingNet->execute_forward_backward("forward", {noiseTimeSeries, delayedSignalTensor});
162
163 if (results.error() != executorch::runtime::Error::Ok)
164 {
165 fprintf(stderr, "Failed to execute forward_backward");
166 return 0;
167 }
168 if (learningIsOn)
169 {
170 optimizer->step(trainingNet->named_gradients("forward").get());
171 }
172
173 remover = results.get()[1].toTensor().const_data_ptr<float>()[0];
174 f_nn = delayed_signal - remover;
175
176 return f_nn;
177 }
